    \ifx{#1}{Undefined Theorem Name}
    \renewcommand\nm@thmtype{theorem}
    \else\renewcommand\thmsname{#1}
    \renewcommand\nm@thmtype{namedtheorem}\def\@currentlabelname{#1}
    \fi
    \begin{\nm@thmtype}}
    {\end{\nm@thmtype}}

    \fi
    \begin{\nm@thmtype}}
